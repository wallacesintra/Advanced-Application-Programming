```c#

// Define the array
int[] numbers = new int[4];

// Insert numbers using a loop
for (int i = 0; i < numbers.Length; i++)
{
    numbers[i] = i + 1; // Assign values (1, 2, 3, 4)
}

// Print the array
Console.WriteLine("Array: ");
foreach (int num in numbers)
{
    Console.Write(num + " ");
}
```
